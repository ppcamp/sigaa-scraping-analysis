41  def __indent__(self):
42  """ Used to parse tags"""
43  return '\t'*len(self.__stack__)

45  def create_tag(self, tagname, value):
46  """ Create a new tag
47  @param tagname Name desired for xml's tags.
48  @param value Value to be placed inside,an empty value '', means that will have a break
49  """
50  self.__out__ += self.__indent__() + '<{}>'.format(tagname)
51  self.__stack__.append(tagname)
52 
53  if value == '':
54  self.__out__ += '\n'
55  else:
56  self.__out__ += value
57  self.end_tag()
58 
59  def end_tag(self):
60  """ Pop's out from stack, and close last tag"""
61  tagname = self.__stack__.pop()
62  if self.__out__[-1] == '\n':
63  self.__out__ += self.__indent__()
64  self.__out__ += '</{}>\n'.format(tagname)
